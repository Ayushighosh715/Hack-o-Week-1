<script>
let tasks = [
  { id:1, text:'Review project proposal', cat:'work', done:false, priority:'high' },
  { id:2, text:'Buy groceries', cat:'personal', done:false, priority:'med' },
  { id:3, text:'Fix login page bug', cat:'urgent', done:false, priority:'high' },
  { id:4, text:'Read chapter 5 of Clean Code', cat:'personal', done:true, priority:'low' },
  { id:5, text:'Prepare weekly standup notes', cat:'work', done:false, priority:'med' },
  { id:6, text:'New app idea: offline journaling', cat:'idea', done:false, priority:'low' },
];
let nextId = 7;
let currentFilter = 'all';

const priorities = { urgent:'high', work:'med', personal:'low', idea:'low' };

function addTask() {
  const input = document.getElementById('taskInput');
  const cat = document.getElementById('catSelect').value;
  const text = input.value.trim();
  if (!text) { input.focus(); return; }
  tasks.unshift({ id: nextId++, text, cat, done: false, priority: priorities[cat] });
  input.value = '';
  render();
}

function toggleDone(id) {
  const t = tasks.find(t => t.id === id);
  if (t) t.done = !t.done;
  render();
}

function deleteTask(id) {
  tasks = tasks.filter(t => t.id !== id);
  render();
}

function filter(f, btn) {
  currentFilter = f;
  document.querySelectorAll('.filter-tab').forEach(b => b.classList.remove('active'));
  btn.classList.add('active');
  render();
}

function render() {
  let visible = tasks;
  if (currentFilter === 'active') visible = tasks.filter(t => !t.done);
  else if (currentFilter === 'done') visible = tasks.filter(t => t.done);
  else if (['work','personal','urgent','idea'].includes(currentFilter)) visible = tasks.filter(t => t.cat === currentFilter);

  const total = tasks.length;
  const done = tasks.filter(t => t.done).length;
  document.getElementById('totalCount').textContent = total;
  document.getElementById('doneCount').textContent = done;
  document.getElementById('leftCount').textContent = total - done;
  const pct = total ? Math.round((done/total)*100) : 0;
  document.getElementById('progressFill').style.width = pct + '%';
  document.getElementById('progressLabel').textContent = pct + '%';

  const list = document.getElementById('taskList');
  if (!visible.length) {
    list.innerHTML = '<div class="empty"><span class="empty-icon">EMPTY</span><p>No tasks here. Add something.</p></div>';
    return;
  }

  const active = visible.filter(t => !t.done);
  const doneItems = visible.filter(t => t.done);

  list.innerHTML = '';
  if (active.length) {
    list.innerHTML += `<div class="date-group">
      <div class="date-label">Active <span class="date-count">${active.length} tasks</span></div>
      ${active.map(taskHTML).join('')}
    </div>`;
  }
  if (doneItems.length) {
    list.innerHTML += `<div class="date-group">
      <div class="date-label">Completed <span class="date-count">${doneItems.length} tasks</span></div>
      ${doneItems.map(taskHTML).join('')}
    </div>`;
  }
}

function taskHTML(t) {
  return `
    <div class="task ${t.done ? 'done' : ''}">
      <div class="task-priority priority-${t.priority}"></div>
      <div class="task-checkbox" onclick="toggleDone(${t.id})">
        <input type="checkbox" ${t.done ? 'checked' : ''} onchange="toggleDone(${t.id})" onclick="event.stopPropagation()">
      </div>
      <div class="task-body">
        <div class="task-text">${t.text}</div>
        <span class="task-cat cat-${t.cat}">${t.cat}</span>
      </div>
      <button class="task-delete" onclick="deleteTask(${t.id})" title="Delete">✕</button>
    </div>
  `;
}

render();
</script>
